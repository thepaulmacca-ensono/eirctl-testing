=== Virtual WAN with Connected Spoke

Deploy Virtual WAN and connect a spoke VNet:

[source,hcl]
----
module "vwan" {
  source   = "./src/connectivity-landing-zone-virtual-wan/deploy/terraform"
  location = "uksouth"
}

resource "azurerm_virtual_network" "spoke" {
  name                = "vnet-spoke-workload"
  address_space       = ["10.10.0.0/16"]
  location            = "uksouth"
  resource_group_name = "rg-workload"
}

resource "azurerm_virtual_hub_connection" "spoke" {
  name                      = "conn-spoke-workload"
  virtual_hub_id            = module.vwan.virtual_hub_id
  remote_virtual_network_id = azurerm_virtual_network.spoke.id
}
----
